# Nueva fecha en formato "dd/mm/yyyy"
import dateparser

def parse_fecha(texto):
    if isinstance(texto, str):
        fecha = dateparser.parse(texto, languages=['es', 'en'])
        if fecha:
            return fecha.strftime('%d/%m/%Y')
    return None

# Lematización
def lematizar_tokens(tokens):
    doc = nlp(' '.join(tokens))
    lemmas_limpios = []
    for token in doc:
        # obtenemos el lema
        lema = token.lemma_.lower().strip()
        lema = re.sub(r'[^a-záéíóúüñ0-9]', '', lema)  # # como precaución quitamos todo lo que no sea letras/números, porque usamos la paquería spacy
        if lema and len(lema) > 1:  # Filtra lemas vacíos o de un solo carácter
            lemmas_limpios.append(lema)
    return lemmas_limpios

#Código de sentimientos
def detectar_sentimiento_manual(fila):
    lemas = fila['lemmas_titulo'] + fila['lemmas_descripcion']
    score = 0
    palabras_detectadas = []

    for lema in lemas:
        lema = lema.lower()
        if lema in palabras_positivas:
            score += 1
            palabras_detectadas.append((lema, 'positivo'))
        elif lema in palabras_negativas:
            score -= 1
            palabras_detectadas.append((lema, 'negativo'))

    if score > 0:
        sentimiento = 'positivo'
    elif score < 0:
        sentimiento = 'negativo'
    else:
        sentimiento = 'neutral'

    return {
        'sentimiento': sentimiento,
        'score': score,
        'palabras_detectadas': palabras_detectadas
    }
